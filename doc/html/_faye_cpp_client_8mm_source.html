168 id FayeCppDelegateWrapper::objcObject(const Variant & variant)
169 {
170  switch (variant.type())
171  {
172  case Variant::TypeInteger:
173  return [NSNumber numberWithLongLong:variant.toInt64()];
174  break;
175  case Variant::TypeUnsignedInteger:
176  return [NSNumber numberWithUnsignedLongLong:variant.toUInt64()];
177  break;
178  case Variant::TypeReal:
179  return [NSNumber numberWithDouble:variant.toDouble()];
180  break;
181  case Variant::TypeBool:
182  return [NSNumber numberWithBool:(BOOL)variant.toBool()];
183  break;
184  case Variant::TypeString:
185  return FayeCppDelegateWrapper::objcString(variant.toString());
186  break;
187  case Variant::TypeMap:
188  return FayeCppDelegateWrapper::objcDict(variant.toMap());
189  break;
190  case Variant::TypeList:
191  return FayeCppDelegateWrapper::objcArray(variant.toList());
192  break;
193  default:
194  break;
195  }
196  return [NSNull null];
197 }

228 Variant FayeCppDelegateWrapper::objcObjectToVariant(id object)
229 {
230  if ([object isKindOfClass:[NSDictionary class]])
231  {
232  VariantMap m;
233  FayeCppDelegateWrapper::objcDictToMap((NSDictionary *)object, m);
234  return Variant(m);
235  }
236  else if ([object isKindOfClass:[NSArray class]])
237  {
238  VariantList l;
239  for (id obj in (NSArray*)object)
240  {
241  l.add(FayeCppDelegateWrapper::objcObjectToVariant(obj));
242  }
243  return Variant(l);
244  }
245  else if ([object isKindOfClass:[NSString class]])
246  {
247  return Variant([(NSString*)object UTF8String]);
248  }
249  else if ([object isKindOfClass:[NSNumber class]])
250  {
251  NSNumber * num = (NSNumber *)object;
252  if (strcmp([num objCType], @encode(BOOL)) == 0) return Variant((bool)[num boolValue]);
253  else
254  {
255  switch (CFNumberGetType((CFNumberRef)num)) {
256  case kCFNumberFloatType:
257  case kCFNumberDoubleType:
258  case kCFNumberCGFloatType:
259  case kCFNumberFloat32Type:
260  case kCFNumberFloat64Type:
261  return Variant((double)[num doubleValue]);
262  break;
263 
264  default:
265  return Variant((long long)[num longLongValue]);
266  break;
267  }
268  }
269  }
270  return Variant();
271 }
