192  bool cancel()
193  {
194 #if defined(HAVE_PTHREAD_H)
195  if (_reThreadThread)
196  {
197  bool res = false;
198  pthread_t th = _reThreadThread;
199 
200  pthread_kill(th, SIGUSR2);
201 
202 #if defined(HAVE_FUNCTION_PTHREAD_CANCEL)
203  if (pthread_cancel(th) == 0)
204 #else
205  pthread_kill(th, SIGUSR2);
206 #endif
207  {
208  void * r = NULL;
209  if (pthread_join(th, &r) == 0)
210  {
211 #if defined(HAVE_FUNCTION_PTHREAD_CANCEL)
212  res = (r == PTHREAD_CANCELED);
213 #else
214  res = true;
215 #endif
216  }
217  }
218  _reThreadThread = (pthread_t)0;
219  return res;
220  }
221 #elif defined (__RE_USING_WINDOWS_THREADS__)
222  if (_reThreadThread)
223  {
224  bool isAlive = false;
225  DWORD dwExitCode = 0;
226  if (GetExitCodeThread(_reThreadThread, &dwExitCode))
227  {
228  // if return code is STILL_ACTIVE,
229  // then thread is live.
230  isAlive = (dwExitCode == STILL_ACTIVE);
231  }
232  if (isAlive)
233  {
234  TerminateThread(_reThreadThread, 0);
235  }
236  CloseHandle(_reThreadThread);
237  _reThreadThread = (HANDLE)0;
238  return true;
239  }
240 #endif
241  return true;
242  }

244  float priority() const
245  {
246 #if defined(HAVE_PTHREAD_H)

258  return ((float)p.sched_priority / (float)maxPrior);

261  }
262 #elif defined(__RE_USING_WINDOWS_THREADS__)

268  return ( (float)(prior - minPriority) / (float)range );

325  {
