206  void REMutableString::replaceWithLen(const char * charsStringValue,
207  const char * withCharsStringValue,
208  const REUInt32 firstLen,
209  const REUInt32 secondLen)
210  {
211  if (firstLen)
212  {
213  const REUInt32 length = this->length();
214  REUInt32 ocurencesCount = 0;
215  char * mainString = (char *)_p->buffer();
216  while ( (mainString = strstr(mainString, charsStringValue)) )
217  {
218  ocurencesCount++;
219  mainString += firstLen;
220  }
221 
222  if (ocurencesCount)
223  {
224  const REUInt32 resultLen = (length + (ocurencesCount * secondLen)) - (ocurencesCount * firstLen);
225  REBuffer * newBuffer = new REBuffer();
226  if (!newBuffer) return;
227  if (!newBuffer->resize(resultLen + 1, false)) return;
228 
229  char * newData = (char *)newBuffer->buffer();
230  mainString = (char *)_p->buffer();
231  char * srcString = (char *)_p->buffer();
232  char * dstString = newData;
233  while ( (mainString = strstr(mainString, charsStringValue)) )
234  {
235  while (srcString != mainString)
236  {
237  *dstString++ = *srcString++;
238  }
239  REUByte * addString = (REUByte *)withCharsStringValue;
240  REUInt32 addLen = secondLen;
241  while (addLen)
242  {
243  *dstString++ = *addString++;
244  addLen--;
245  }
246  mainString += firstLen;
247  srcString += firstLen;
248  }
249  while (*srcString)
250  {
251  *dstString++ = *srcString++;
252  }
253  newData[resultLen] = 0;
254  _p = REPtr<REBuffer>(newBuffer);
255  }
256  }
