117  void Client::onClientResponceMessageReceived(const VariantMap & message)
118  {
119  VariantMap::Iterator i = message.iterator();
120  while (i.next())
121  {
122  if (i.key().isEqual("channel"))
123  {
124  REString channel = i.value().toString();
125  if (channel.isEqual(HANDSHAKE_CHANNEL))
126  {
127  this->onHandshakeDone(message);
128  }
129  else if (channel.isEqual(CONNECT_CHANNEL))
130  {
131  this->onConnectFayeDone(message);
132  }
133  else if (channel.isEqual(DISCONNECT_CHANNEL))
134  {
135  this->onDisconnectFayeDone(message);
136  }
137  else if (channel.isEqual(SUBSCRIBE_CHANNEL))
138  {
139  this->onSubscriptionDone(message);
140  }
141  else if (channel.isEqual(UNSUBSCRIBE_CHANNEL))
142  {
143  this->onUnsubscribingDone(message);
144  }
145  else if (_subscribedChannels.isContaines(channel))
146  {
147  this->onReceivedMessageOnChannel(message, channel);
148  }
149  }
150  }
151  }

248  void Client::onHandshakeDone(const VariantMap & message)
249  {
250 #ifdef FAYECPP_DEBUG_MESSAGES
251  RELog::log("Client: onHandshakeDone");
252 #endif
253 
254  VariantMap::Iterator i = message.iterator();
255  while (i.next())
256  {
257  if (i.key().isEqual("clientId") && i.value().isString())
258  {
259  _clientId = i.value().toString();
260  }
261  else if (i.key().isEqual("supportedConnectionTypes") && i.value().isList())
262  {
263  VariantList::Iterator j = i.value().toList().iterator();
264  while (j.next()) _supportedConnectionTypes.add(j.value().toString());
265  }
266  }
267 
268  if (_clientId.isEmpty())
269  {
270  if (_delegate) _delegate->onFayeErrorString(this, REStaticString("Handshake clientId is empty."));
271  return;
272  }
273 
274 #ifdef FAYECPP_DEBUG_MESSAGES
275  RELog::log("Client: clientId=%s", _clientId.UTF8String());
276 #endif
277 
278  if (_supportedConnectionTypes.isEmpty())
279  {
280  if (_delegate) _delegate->onFayeErrorString(this, REStaticString("Handshake supported connection types is empty."));
281  return;
282  }
283 
284  REStringList availableTypes = Client::availableConnectionTypes();
285  const REString currentType = this->currentTransportName();
286  bool isCurrentTypeFound = false;
287  REStringList::Iterator j = _supportedConnectionTypes.iterator();
288  while (!isCurrentTypeFound && j.next())
289  {
290  isCurrentTypeFound = availableTypes.isContaines(j.value()) && currentType.isEqual(j.value());
291  }
292 
293  if (isCurrentTypeFound)
294  {
295  this->connectFaye();
296  this->subscribePendingSubscriptions();
297  }
298  else
299  {
300  if (_delegate)
301  {
302  REMutableString error("Can't find implemented faye transport protocol type from supported by the server: [");
303  unsigned int index = 0;
304  j = _supportedConnectionTypes.iterator();
305  while (i.next())
306  {
307  if (index) error.append(", ");
308  error.append(i.value().toString());
309  index++;
310  }
311  error.append("]");
312  _delegate->onFayeErrorString(this, error);
313  }
314  }
315  }
