68  int64_t Variant::toInt64() const
69  {
70  switch (_t)
71  {
72  case TypeInteger: return _u.int64Value; break;
73  case TypeUnsignedInteger: return (int64_t)_u.uint64Value; break;
74  case TypeReal: return (int64_t)_u.doubleValue; break;
75  case TypeBool: return _u.boolValue ? (int64_t)1 : (int64_t)0; break;
76  case TypeString:
77  {
78  if (this->toString().isDigit())
79  {
80  REBOOL isOk = false;
81  const REInt64 parsedValue = this->toString().integerValue(&isOk);
82  if (isOk) return (int64_t)parsedValue;
83  }
84  }
85  break;
86  default: break;
87  }
88  return 0;
89  }

137  bool Variant::toBool() const
138  {
139  switch (_t)
140  {
141  case TypeInteger: return _u.int64Value == 0 ? false : true; break;
142  case TypeUnsignedInteger: return _u.uint64Value == 0 ? false: true; break;
143  case TypeReal: return _u.doubleValue == 0 ? false : true; break;
144  case TypeBool: return _u.boolValue; break;
145  case TypeString:
146  {
147  REMutableString s = this->toString().mutableString();
148  s.toLower();
149  if (s.isEqual("true")) return true;
150  }
151  break;
152  default: break;
153  }
154  return false;
155  }
