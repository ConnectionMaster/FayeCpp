83  int WebSocket::callbackEcho(struct libwebsocket_context * context,
84  struct libwebsocket * wsi,
85  enum libwebsocket_callback_reasons reason,
86  void * user,
87  void * input,
88  size_t len)
89  {
90  WebSocket * socket = static_cast<WebSocket *>(libwebsocket_context_user(context));
91  EchoSessionData * pss = static_cast<EchoSessionData *>(user);
92  switch (reason)
93  {
94  case LWS_CALLBACK_CLIENT_ESTABLISHED:
95  socket->onCallbackEstablished();
96  break;
97 
98  case LWS_CALLBACK_CLIENT_RECEIVE:
99  if (input && len) socket->onCallbackReceive(input, len);
100  break;
101 
102  case LWS_CALLBACK_CLIENT_WRITEABLE:
103  return socket->onCallbackWritable(context, wsi, pss);
104  break;
105 
106  default:
107  break;
108  }
109  return 0;
110  }

129  int WebSocket::onCallbackWritable(struct libwebsocket_context * context,
130  struct libwebsocket * connection,
131  EchoSessionData * pss)
132  {
133  WriteBuffer * buffer = NULL;
134 
135  this->writeLock();
136  REList<WriteBuffer *>::Iterator i = _writeBuffers.iterator();
137  while (!buffer && i.next())
138  {
139  buffer = i.value();
140  _writeBuffers.removeNode(i.node());
141  }
142  this->writeUnlock();
143 
144  if (!buffer) return 0;
145 
146  const enum libwebsocket_write_protocol type = (enum libwebsocket_write_protocol)buffer->tag;
147  memcpy(&pss->buf[LWS_SEND_BUFFER_PRE_PADDING], buffer->buffer(), buffer->size());
148  pss->len = buffer->size();
149 
150  delete buffer;
151 
152  this->updateLastSendTime();
153  const int writed = libwebsocket_write(connection, &pss->buf[LWS_SEND_BUFFER_PRE_PADDING], pss->len, type);
154  if (writed < 0)
155  {
156 #ifdef FAYECPP_DEBUG_MESSAGES
157  RELog::log("ERROR %d writing to socket, hanging up", writed);
158 #endif
159  libwebsocket_callback_on_writable(context, connection);
160  return -1;
161  }
162  if (writed < pss->len)
163  {
164 #ifdef FAYECPP_DEBUG_MESSAGES
165  RELog::log("Partial write");
166 #endif
167  libwebsocket_callback_on_writable(context, connection);
168  return -1;
169  }
170 
171  if (!_writeBuffers.isEmpty())
172  {
173  libwebsocket_callback_on_writable(context, connection);
174  }
175 
176  return 0;
177  }

179  void WebSocket::addWriteBufferData(const unsigned char * data, const size_t dataSize, const enum libwebsocket_write_protocol type)
180  {
181  bool isError = false;
182 
183  this->writeLock();
184 
185  WriteBuffer * buffer = new WriteBuffer(data, (REUInt32)dataSize);
186  if (buffer && buffer->size() == dataSize)
187  {
188  buffer->tag = (int)type;
189 #ifdef FAYECPP_DEBUG_MESSAGES
190  RELog::log("WILL WRITE %i bytes: %s", (int)buffer->size(), (char *)buffer->buffer());
191 #endif
192  _writeBuffers.add(buffer);
193  }
194  else
195  {
196  isError = true;
197  }
198 
199  this->writeUnlock();
200

202  {
203  this->onError("Can't send buffer data");
204  }

206  {
207  libwebsocket_callback_on_writable(_context, _connection);
