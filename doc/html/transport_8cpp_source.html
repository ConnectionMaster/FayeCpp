46  void Transport::receivedAdvice(const VariantMap & advice)
47  {
48  Advice a;
49  a.reconnect = ADVICE_RECONNECT_NONE;
50  a.timeout = a.interval = -1;
51 
52  VariantMap::Iterator i = advice.iterator();
53  while (i.next())
54  {
55  if (i.key().isEqual("reconnect") && i.value().isString())
56  {
57  if (i.value().toString().isEqual("retry"))
58  a.reconnect = ADVICE_RECONNECT_RETRY;
59  else if (i.value().toString().isEqual("handshake"))
60  a.reconnect = ADVICE_RECONNECT_HANDSHAKE;
61  }
62  else if (i.key().isEqual("interval") && i.value().isNumber())
63  a.interval = i.value().toInt64() / 1000;
64  else if (i.key().isEqual("timeout") && i.value().isNumber())
65  a.timeout = i.value().toInt64() / 1000;
66  }
67 
68  _advice = a;
69 
70  Variant * thisTransportAdvice = advice.findTypedValue(this->name(), Variant::TypeMap);
71  if (thisTransportAdvice) this->receivedAdvice(thisTransportAdvice->toMap());
72  }

137  void Transport::setUrl(const char * url)
138  {
139  _url = url;
140  REMutableString urlString(url);
141  if (urlString.isContaines("ws://"))
142  {
143  urlString.replace("ws://");
144  _isUseSSL = false;
145  }
146 
147  if (urlString.isContaines("wss://"))
148  {
149  urlString.replace("wss://");
150  _isUseSSL = true;
151  }
152 
153  if (urlString.isContaines("http://"))
154  {
155  urlString.replace("http://");
156 #if defined(HAVE_SUITABLE_QT_VERSION)
157  REMutableString u(url);
158  u.replace("http://", "ws://");
159  _url = u.UTF8String();
160 #endif
161  _isUseSSL = false;
162  }
163 
164  if (urlString.isContaines("https://"))
165  {
166  urlString.replace("https://");
167 #if defined(HAVE_SUITABLE_QT_VERSION)
168  REMutableString u(url);
169  u.replace("https://", "wss://");
170  _url = u.UTF8String();
171 #endif
172  _isUseSSL = true;
173  }
174 
175  const char * sub = strstr(urlString.UTF8String(), ":");
176  if (sub)
177  {
178  int port = -1;
179  if (sscanf(++sub, "%i", &port) == 1)
180  {
181  _port = port;
182  }
183  }
184 
185  sub = strstr(urlString.UTF8String(), "/");
186  if (sub)
187  {
188  _path = sub;
189  }
190  else
191  {
192  _path = "/";
193  }
194 
195  sub = strstr(urlString.UTF8String(), ":");
196  if (!sub) sub = strstr(urlString.UTF8String(), "/");
197  if (sub)
198  {
199  const REUInt32 len = (REUInt32)(sub - urlString.UTF8String());
200  _host = REString(urlString.UTF8String(), len);
201  }
202  else
203  {
204  _host = urlString.UTF8String();
205  }
206  }
