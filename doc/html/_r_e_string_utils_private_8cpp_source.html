196  int REStringUtilsPrivate::wideToChars(const wchar_t * wideString, int wideStringLength, char * charsString)
197  {
198  int stringLength = 0;
199  if (wideStringLength > 0)
200  {
201  int l = wideStringLength;
202  unsigned char * cursor = (unsigned char *)charsString;
203  const wchar_t * ch = wideString;
204  for (int i = 0; i < l; i++)
205  {
206  wchar_t u = *ch;
207  if (u < (unsigned char)0x80)
208  {
209  *cursor++ = (unsigned char)u;
210  }
211  else
212  {
213  if (u < 0x0800)
214  {
215  *cursor++ = (unsigned char)0xc0 | ((unsigned char) (u >> 6));
216  }
217  else
218  {
219  if ( (u >= 0xd800) && (u < 0xdc00) && (i < l-1))
220  {
221  unsigned short low = ch[1];
222  if ((low >= 0xdc00) && (low < 0xe000))
223  {
224  ++ch;
225  ++i;
226  u = ((u - 0xd800)*0x400) + (low - 0xdc00) + 0x10000;
227  }
228  }
229  if (u > 0xffff)
230  {
231  // if people are working in utf8, but strings are encoded in eg. latin1, the resulting
232  // name might be invalid utf8. This and the corresponding code in fromUtf8 takes care
233  // we can handle this without loosing information. This can happen with latin filenames
234  // and a utf8 locale under Unix.
235  if ( (u > 0x10fe00) && (u < 0x10ff00) )
236  {
237  *cursor++ = (u - 0x10fe00);
238  ++ch;
239  continue;
240  }
241  else
242  {
243  *cursor++ = (unsigned char)0xf0 | ((unsigned char) (u >> 18));
244  *cursor++ = (unsigned char)0x80 | (((unsigned char) (u >> 12)) & (unsigned char)0x3f);
245  }
246  }
247  else
248  {
249  *cursor++ = (unsigned char)0xe0 | ((unsigned char) (u >> 12));
250  }
251  *cursor++ = (unsigned char)0x80 | (((unsigned char) (u >> 6)) & (unsigned char)0x3f);
252  }
253  *cursor++ = (unsigned char)0x80 | ((unsigned char) (u & (unsigned char)0x3f));
254  }
255  ++ch;
256  }
257  stringLength = (int)(cursor - (unsigned char*)charsString);
258  }
259  return stringLength;
260  }

262  int REStringUtilsPrivate::charsToWide(const char * charsString, int charsStringLengthOrNegative, wchar_t * wideString)
263  {
264  if ( charsStringLengthOrNegative < 0)
265  {
266  charsStringLengthOrNegative = (int)strlen(charsString);
267  }
268 
269  wchar_t uc = 0;
270  int need = 0;
271  int error = -1;
272  unsigned char ch;
273  int stringLength = 0;
274  for (int i = 0; i < charsStringLengthOrNegative; ++i)
275  {
276  ch = charsString[i];
277  if (need)
278  {
279  if ((ch&(unsigned char)0xc0) == (unsigned char)0x80)
280  {
281  uc = (uc << 6) | (ch & (unsigned char)0x3f);
282  need--;
283  if (!need)
284  {
285  if (uc > 0xffff)
286  {
287  uc -= 0x10000;
288  wchar_t high = (uc/0x400) + 0xd800;
289  wchar_t low = (uc%0x400) + 0xdc00;
290  *wideString++ = high;
291  *wideString++ = low;
292  stringLength++;
293  }
294  else
295  {
296  *wideString++ = uc;
297  stringLength++;
298  }
299  }
300  }
301  else
302  {
303  i = error;
304  *wideString++ = (wchar_t)0xdbff;
305  *wideString++ = (wchar_t)0xde00 + ((unsigned char)charsString[i]);
306  stringLength++;
307  need = 0;
308  }
309  }
310  else
311  {
312  if (ch < 128)
313  {
314  *wideString++ = ch;
315  stringLength++;
316  }
317  else if ((ch & 0xe0) == 0xc0)
318  {
319  uc = ch & 0x1f;
320  need = 1;
321  error = i;
322  }
323  else if ((ch & 0xf0) == 0xe0)
324  {
325  uc = ch & 0x0f;
326  need = 2;
327  error = i;
328  }
329  else if ((ch & 0xf8) == 0xf0)
330  {
331  uc = ch & 0x07;
332  need = 3;
333  error = i;
334  }
335  else
336  {
337  *wideString++ = 0xdbff;
338  *wideString++ = 0xde00 + ((unsigned char)charsString[i]);
339  stringLength++;
340  }
341  }
342  }
343  if (need)
344  {
345  for (int i = error; i < charsStringLengthOrNegative; ++i)
346  {
347  *wideString++ = 0xdbff;
348  *wideString++ = 0xde00 + (unsigned char)charsString[i];
349  stringLength++;
350  }
351  }
352  return stringLength;
353  }
