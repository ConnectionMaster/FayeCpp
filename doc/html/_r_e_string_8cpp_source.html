152  REBOOL REString::isDigit() const
153  {
154  const char * ch = this->UTF8String();
155  if (ch)
156  {
157  REUInt32 processed = 0;
158  while (*ch)
159  {
160  const int dig = *ch++;
161  processed++;
162  if (!isdigit(dig))
163  {
164  if ((dig != '+') || (dig != '-') || (dig != '.'))
165  {
166  return false;
167  }
168  }
169  }
170  return (processed > 0);
171  }
172  return false;

309  REString REString::createWithFormat(const char * format, ...)
310  {
311  if (format)
312  {
313  va_list args;
314  va_start(args, format);
315  char strBuff[1024];
316 
317 #if defined(HAVE_FUNCTION_VSNPRINTF)
318  const int writed = vsnprintf(strBuff, 1024, format, args);
319 #elif defined(HAVE_FUNCTION_VSPRINTF_S)
320  const int writed = vsprintf_s(strBuff, format, args);
321 #else
322  const int writed = vsprintf(strBuff, format, args);
323 #endif
324  va_end(args);
325  if (writed > 0)
326  {
327  strBuff[writed] = 0;
328  return REString(strBuff, (REUInt32)writed);

330  }
