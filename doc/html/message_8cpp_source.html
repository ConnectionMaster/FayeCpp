91  Responce & Responce::setMessageData(const unsigned char * data, const size_t dataSize)
92  {
93  if (data && dataSize > 0)
94  {
95  bool isOk = false;
96  JsonParser parser((const char *)data);
97  if (parser.isList())
98  {
99  _messageList = new VariantList();
100  if (_messageList) isOk = parser.toList(*_messageList);
101  }
102  else if (parser.isMap())
103  {
104  _messageMap = new VariantMap();
105  if (_messageMap) isOk = parser.toMap(*_messageMap);
106  }
107 
108  if (!isOk)
109  {
110  _messageBuffer = new REBuffer(data, (REUInt32)dataSize);
111  }
112  }
113  return *this;
114  }
